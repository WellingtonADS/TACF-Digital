import Button from "@/components/ui/Button";
import { X, MapPin, Clock, Mail, Phone, Calendar as CalendarIcon, CheckCircle2, Sun, Moon, Info } from "lucide-react";
import { format, parseISO } from "date-fns";
import { ptBR } from "date-fns/locale";
import { useState, useEffect } from "react";
import { Session } from "@/types/database.types";
import { cn } from "@/lib/utils"; // Assumindo que você tenha um utilitário de classes ou use string simples

interface BookingConfirmationModalProps {
  isOpen: boolean;
  onClose: () => void;
  onConfirm: (sessionId: string, tafType: "1" | "2") => void;
  date: string | null;            // A data clicada
  availableSessions: Session[];   // As sessões disponíveis naquele dia (Manhã e/ou Tarde)
  loading?: boolean;
}

export default function BookingConfirmationModal({
  isOpen,
  onClose,
  onConfirm,
  date,
  availableSessions,
  loading
}: BookingConfirmationModalProps) {
  // Estados de Seleção
  const [selectedTaf, setSelectedTaf] = useState<"1" | "2">("1");
  const [selectedPeriod, setSelectedPeriod] = useState<"morning" | "afternoon" | null>(null);

  // Resetar estados quando o modal abre/fecha
  useEffect(() => {
    if (isOpen) {
      setSelectedTaf("1"); // Padrão
      setSelectedPeriod(null);
    }
  }, [isOpen, date]);

  if (!isOpen || !date) return null;

  const dateObj = parseISO(date);
  const formattedDate = format(dateObj, "EEEE, dd 'de' MMMM 'de' yyyy", { locale: ptBR });
  const displayDate = formattedDate.charAt(0).toUpperCase() + formattedDate.slice(1);

  // Verifica quais turnos existem para essa data
  const hasMorning = availableSessions.some(s => s.period === "morning");
  const hasAfternoon = availableSessions.some(s => s.period === "afternoon");

  const handleConfirm = () => {
    if (!selectedPeriod) return;
    
    // Encontra a sessão correspondente ao turno escolhido
    const session = availableSessions.find(s => s.period === selectedPeriod);
    if (session) {
      onConfirm(session.id, selectedTaf);
    }
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-slate-900/60 backdrop-blur-sm animate-in fade-in">
      <div className="bg-white rounded-3xl shadow-2xl max-w-lg w-full overflow-hidden animate-in zoom-in-95 duration-200 flex flex-col max-h-[90vh]">
        
        {/* Cabeçalho */}
        <div className="bg-slate-50 px-8 py-6 border-b border-slate-100 flex justify-between items-start">
          <div>
            <h2 className="text-2xl font-bold text-slate-800">Confirmar Presença</h2>
            <p className="text-slate-500 text-sm mt-1 flex items-center gap-2">
              <CalendarIcon size={14} /> {displayDate}
            </p>
          </div>
          <button onClick={onClose} className="text-slate-400 hover:text-slate-600 transition-colors">
            <X size={24} />
          </button>
        </div>

        {/* Corpo Scrollável */}
        <div className="p-8 overflow-y-auto space-y-8">
          
          {/* SELEÇÃO 1: TAF (Semestre) */}
          <div className="space-y-3">
            <label className="text-sm font-bold text-slate-700 uppercase tracking-wider flex items-center gap-2">
              1. Selecione a Referência
              <Info size={14} className="text-slate-400" />
            </label>
            <div className="grid grid-cols-2 gap-4">
              <button
                onClick={() => setSelectedTaf("1")}
                className={cn(
                  "relative p-4 rounded-xl border-2 text-left transition-all",
                  selectedTaf === "1" 
                    ? "border-blue-600 bg-blue-50/50 ring-1 ring-blue-600/20" 
                    : "border-slate-100 hover:border-slate-200 bg-white"
                )}
              >
                <span className="block text-lg font-bold text-slate-900">TAF 1</span>
                <span className="text-xs text-slate-500 font-medium">1º Semestre</span>
                {selectedTaf === "1" && <CheckCircle2 className="absolute top-4 right-4 text-blue-600 w-5 h-5" />}
              </button>

              <button
                onClick={() => setSelectedTaf("2")}
                className={cn(
                  "relative p-4 rounded-xl border-2 text-left transition-all",
                  selectedTaf === "2" 
                    ? "border-blue-600 bg-blue-50/50 ring-1 ring-blue-600/20" 
                    : "border-slate-100 hover:border-slate-200 bg-white"
                )}
              >
                <span className="block text-lg font-bold text-slate-900">TAF 2</span>
                <span className="text-xs text-slate-500 font-medium">2º Semestre</span>
                {selectedTaf === "2" && <CheckCircle2 className="absolute top-4 right-4 text-blue-600 w-5 h-5" />}
              </button>
            </div>
          </div>

          {/* SELEÇÃO 2: TURNO (Manhã/Tarde) */}
          <div className="space-y-3">
            <label className="text-sm font-bold text-slate-700 uppercase tracking-wider">
              2. Escolha o Turno
            </label>
            <div className="space-y-3">
              {/* Opção Manhã */}
              <button
                onClick={() => hasMorning && setSelectedPeriod("morning")}
                disabled={!hasMorning}
                className={cn(
                  "w-full flex items-center p-4 rounded-xl border-2 transition-all group",
                  !hasMorning ? "opacity-50 cursor-not-allowed bg-slate-50 border-slate-100" :
                  selectedPeriod === "morning"
                    ? "border-blue-600 bg-blue-50/50"
                    : "border-slate-100 hover:border-blue-200 bg-white"
                )}
              >
                <div className={cn("p-3 rounded-lg mr-4", selectedPeriod === "morning" ? "bg-blue-100 text-blue-600" : "bg-slate-100 text-slate-500")}>
                  <Sun size={24} />
                </div>
                <div className="text-left flex-1">
                  <span className="block font-bold text-slate-900">Turno da Manhã</span>
                  <span className="text-sm text-slate-500">Início às 08:00hs</span>
                </div>
                {!hasMorning && <span className="text-xs font-bold text-red-400 px-2 py-1 bg-red-50 rounded">Indisponível</span>}
                {selectedPeriod === "morning" && <CheckCircle2 className="text-blue-600 w-6 h-6" />}
              </button>

              {/* Opção Tarde */}
              <button
                onClick={() => hasAfternoon && setSelectedPeriod("afternoon")}
                disabled={!hasAfternoon}
                className={cn(
                  "w-full flex items-center p-4 rounded-xl border-2 transition-all group",
                  !hasAfternoon ? "opacity-50 cursor-not-allowed bg-slate-50 border-slate-100" :
                  selectedPeriod === "afternoon"
                    ? "border-blue-600 bg-blue-50/50"
                    : "border-slate-100 hover:border-blue-200 bg-white"
                )}
              >
                <div className={cn("p-3 rounded-lg mr-4", selectedPeriod === "afternoon" ? "bg-blue-100 text-blue-600" : "bg-slate-100 text-slate-500")}>
                  <Moon size={24} />
                </div>
                <div className="text-left flex-1">
                  <span className="block font-bold text-slate-900">Turno da Tarde</span>
                  <span className="text-sm text-slate-500">Início às 13:30hs</span>
                </div>
                {!hasAfternoon && <span className="text-xs font-bold text-red-400 px-2 py-1 bg-red-50 rounded">Indisponível</span>}
                {selectedPeriod === "afternoon" && <CheckCircle2 className="text-blue-600 w-6 h-6" />}
              </button>
            </div>
          </div>

          {/* Info Estática */}
          <div className="bg-slate-50 p-4 rounded-xl border border-slate-100 space-y-2 text-sm text-slate-600">
            <div className="flex items-center gap-2">
              <MapPin size={16} className="text-primary"/> 
              <span className="font-semibold text-slate-800">Local:</span> HAAF - Pista de Atletismo
            </div>
            <div className="flex items-center gap-2">
               <Mail size={16} className="text-primary"/> 
               <span className="font-semibold text-slate-800">Dúvidas:</span> edfisica.haaf@fab.mil.br
            </div>
          </div>

        </div>

        {/* Rodapé */}
        <div className="p-6 border-t border-slate-100 flex gap-3 bg-white">
          <Button 
            variant="outline" 
            onClick={onClose}
            disabled={loading}
            className="flex-1 h-12 text-base"
          >
            Cancelar
          </Button>
          <Button 
            onClick={handleConfirm} 
            isLoading={loading}
            disabled={!selectedPeriod}
            className="flex-[2] h-12 text-base bg-[#1B365D] hover:bg-[#1B365D]/90 shadow-lg shadow-blue-900/20"
          >
            Confirmar Agendamento
          </Button>
        </div>

      </div>
    </div>
  );
}

2. src/pages/UserDashboard.tsx

Lógica ajustada: O clique no calendário captura a Data, filtra as sessões daquele dia e abre o Modal para o usuário decidir o resto.
TypeScript

import BookingConfirmationModal from "@/components/Booking/BookingConfirmationModal";
import ComprovanteTicket from "@/components/Booking/ComprovanteTicket";
import CalendarGrid from "@/components/Calendar/CalendarGrid";
import { Body, H1 } from "@/components/ui/Typography";
import { useAuth } from "@/contexts/AuthContext";
import { getActiveBooking, confirmBooking, fetchFutureSessions } from "@/services/api"; // Importar fetchFutureSessions
import { upsertProfile } from "@/services/supabase"; // Para atualizar o Semestre se necessário
import type { BookingWithDetails, Session } from "@/types/database.types";
import React, { useEffect, useState } from "react";
import { Loader2, AlertTriangle, ArrowRight, History } from "lucide-react";
import { Link } from "react-router-dom";
import { toast } from "sonner";
import { Card } from "@/components/ui/Card";
import { format, parseISO } from "date-fns";
import { ptBR } from "date-fns/locale";

const UserDashboard: React.FC = () => {
  const { user, profile, loading: authLoading } = useAuth();
  
  const [booking, setBooking] = useState<BookingWithDetails | null>(null);
  const [allSessions, setAllSessions] = useState<Session[]>([]); // Todas as sessões carregadas
  const [dataLoading, setDataLoading] = useState(true);
  
  // Estados para o Modal
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [selectedDate, setSelectedDate] = useState<string | null>(null);
  const [daySessions, setDaySessions] = useState<Session[]>([]); // Sessões filtradas do dia
  const [confirmLoading, setConfirmLoading] = useState(false);

  // Verifica perfil
  const isProfileComplete = !!(
    profile?.full_name && 
    profile?.rank && 
    profile?.saram && 
    profile?.war_name && 
    profile?.sector
  );

  useEffect(() => {
    if (!user?.id) return;
    let mounted = true;
    
    const fetchData = async () => {
        try {
            // 1. Busca agendamento ativo
            const b = await getActiveBooking(user.id);
            if (mounted && b) setBooking(b as any);

            // 2. Busca sessões futuras para popular o calendário
            // Nota: O CalendarGrid pode buscar seus próprios dados, mas se trouxermos para cá,
            // temos controle total. Vamos assumir que fetchFutureSessions retorna a lista.
            const sessionsRes = await fetchFutureSessions();
            if (mounted && sessionsRes.data) {
                setAllSessions(sessionsRes.data as unknown as Session[]);
            }

        } catch (error) {
            console.error(error);
        } finally {
            if (mounted) setDataLoading(false);
        }
    };

    fetchData();
    return () => { mounted = false; };
  }, [user?.id]);

  // Handler: Quando clica em uma data no Calendário
  // O CalendarGrid deve emitir esse evento passando a string da data (yyyy-MM-dd)
  // ou um objeto de sessão. Vamos adaptar para receber a data.
  const handleDateSelect = (date: string) => {
    if (booking) {
        toast.error("Você já possui um agendamento ativo.");
        return;
    }
    if (!isProfileComplete) {
        toast.error("Complete seu cadastro antes de agendar.");
        return;
    }

    // Filtra as sessões disponíveis para aquela data
    const sessionsForDay = allSessions.filter(s => s.date === date && s.status === 'open');
    
    if (sessionsForDay.length === 0) {
        toast.info("Não há turnos disponíveis nesta data.");
        return; // Ou abre o modal mesmo assim para mostrar indisponível? Melhor não.
    }

    setSelectedDate(date);
    setDaySessions(sessionsForDay);
    setIsModalOpen(true);
  };

  // Handler: Confirmação vinda do Modal
  const handleConfirmBooking = async (sessionId: string, tafType: "1" | "2") => {
    if (!user) return;
    setConfirmLoading(true);

    try {
        // Opcional: Atualizar o perfil com o semestre escolhido se for relevante para o negócio
        if (profile?.semester !== tafType) {
             await upsertProfile({ id: user.id, semester: tafType } as any);
        }

        // Confirma o agendamento
        const result = await confirmBooking(sessionId);
        
        if (result.success) {
            toast.success("Agendamento Confirmado!");
            setIsModalOpen(false);
            window.location.reload(); 
        } else {
            toast.error(result.error || "Erro ao agendar.");
        }
    } catch (error) {
        toast.error("Erro de conexão.");
    } finally {
        setConfirmLoading(false);
    }
  };

  if (authLoading || dataLoading) {
    return (
      <div className="flex flex-col items-center justify-center min-h-[400px] text-slate-400">
        <Loader2 className="w-8 h-8 animate-spin mb-2" />
        <p className="text-sm">Carregando...</p>
      </div>
    );
  }

  return (
    <div className="space-y-6 animate-in fade-in slide-in-from-bottom-4 duration-700 pb-12">
      
      {/* Header */}
      <div className="flex flex-col gap-1">
        <H1 className="text-3xl">
          Olá, {profile?.war_name || profile?.full_name?.split(" ")[0] || "Militar"}
        </H1>
        <Body className="text-slate-500">
          Consulte a disponibilidade e realize seu TAF.
        </Body>
      </div>

      {/* Alerta de Perfil Incompleto */}
      {!isProfileComplete && (
        <div className="bg-orange-50 border border-orange-200 rounded-xl p-4 flex flex-col sm:flex-row items-start sm:items-center justify-between gap-4">
          <div className="flex items-start gap-3">
             <div className="p-2 bg-orange-100 text-orange-600 rounded-lg shrink-0">
               <AlertTriangle size={20} />
             </div>
             <div>
               <h4 className="font-bold text-orange-800 text-sm">Cadastro Pendente</h4>
               <p className="text-orange-700 text-xs mt-1">
                 Complete seus dados (SARAM, Setor) para agendar.
               </p>
             </div>
          </div>
          <Link 
            to="/profile" 
            className="text-sm font-bold text-orange-700 hover:text-orange-900 flex items-center gap-1 whitespace-nowrap bg-white/50 px-3 py-2 rounded-lg border border-orange-200"
          >
            Completar Agora <ArrowRight size={16} />
          </Link>
        </div>
      )}

      {/* LAYOUT PRINCIPAL */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
        
        {/* COLUNA ESQUERDA: Calendário (2/3) */}
        <div className="lg:col-span-2 space-y-6">
            {booking ? (
                <ComprovanteTicket booking={booking} />
            ) : (
                // Importante: CalendarGrid agora precisa de uma prop 'onDateSelect'
                // Se ele ainda usa 'onSessionSelect', você precisará ajustá-lo para passar a data.
                // Aqui estou assumindo que ele já sabe listar sessões. 
                // Para simplificar, passamos as sessões para ele renderizar os dias ocupados.
                <CalendarGrid
                    sessions={allSessions} // Passamos sessões para ele pintar os dias
                    onDateSelect={handleDateSelect} // Novo manipulador que recebe a DATA
                    isAdmin={false}
                />
            )}
        </div>

        {/* COLUNA DIREITA: Resumo (1/3) */}
        <div className="space-y-6">
            <Card className="h-full border border-slate-200 bg-white p-6 shadow-sm">
                <div className="flex items-center gap-2 mb-6 text-slate-800">
                    <History className="w-5 h-5 text-primary" />
                    <h3 className="font-bold text-lg">Meus Agendamentos</h3>
                </div>

                <div className="space-y-6 relative">
                    <div className="absolute left-[7px] top-2 bottom-2 w-[2px] bg-slate-100" />

                    {booking ? (
                        <div className="relative pl-6">
                            <div className="absolute left-0 top-1.5 w-4 h-4 rounded-full bg-green-500 border-4 border-white shadow-sm" />
                            <div className="text-xs font-bold text-slate-400 uppercase tracking-wider mb-1">Próximo TAF</div>
                            <div className="font-medium text-slate-900">
                                {format(parseISO(booking.session.date), "dd 'de' MMMM 'de' yyyy", { locale: ptBR })}
                            </div>
                            <div className="text-sm text-slate-500 mt-0.5 capitalize">
                                Turno: {booking.session.period === "morning" ? "Manhã" : "Tarde"}
                            </div>
                            <div className="text-xs text-blue-600 font-bold mt-2 bg-blue-50 inline-block px-2 py-1 rounded border border-blue-100">
                                Agendado
                            </div>
                        </div>
                    ) : (
                        <div className="relative pl-6 opacity-60">
                            <div className="absolute left-0 top-1.5 w-4 h-4 rounded-full bg-slate-300 border-4 border-white" />
                            <p className="text-sm text-slate-500 italic">
                                Você não possui agendamentos ativos.
                            </p>
                        </div>
                    )}
                </div>
            </Card>
        </div>

      </div>

      <BookingConfirmationModal 
        isOpen={isModalOpen}
        onClose={() => setIsModalOpen(false)}
        onConfirm={handleConfirmBooking}
        date={selectedDate}
        availableSessions={daySessions}
        loading={confirmLoading}
      />

    </div>
  );
};

export default UserDashboard;
